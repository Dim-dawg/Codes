<script>
// ── State ────────────────────────────────────────────────────
const MONTH_YEAR = getCurrentMonthYear_();
let chatHistory  = [];
let isLoading    = false;
let summaryLoaded = false;
let goalsLoaded   = false;

// ── Init ─────────────────────────────────────────────────────
document.getElementById('monthBadge').textContent = MONTH_YEAR;
document.getElementById('welcomeTime').textContent = formatTime_();
autoResizeTextarea_(document.getElementById('userInput'));

// Load header stats after a short delay
setTimeout(loadHeaderStats_, 600);

// ── Tab switching ─────────────────────────────────────────────
document.querySelectorAll('.tab-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    const tab = btn.dataset.tab;
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
    btn.classList.add('active');
    document.getElementById('panel-' + tab).classList.add('active');

    if (tab === 'summary' && !summaryLoaded) loadSummary_();
    if (tab === 'goals'   && !goalsLoaded)   loadGoals_();
  });
});

// ── Send message ──────────────────────────────────────────────
document.getElementById('sendBtn').addEventListener('click', sendMessage_);
document.getElementById('userInput').addEventListener('keydown', e => {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage_(); }
});

function askQuick(text) {
  // Switch to chat tab
  document.querySelectorAll('.tab-btn')[0].click();
  document.getElementById('userInput').value = text;
  sendMessage_();
}

function sendMessage_() {
  const input = document.getElementById('userInput');
  const query = input.value.trim();
  if (!query || isLoading) return;

  appendUserMessage_(query);
  input.value = '';
  input.style.height = 'auto';
  chatHistory.push({ sender: 'User', text: query });

  const historyStr = chatHistory.slice(-6)
    .map(m => `${m.sender}: ${m.text}`)
    .join('\n');

  setLoading_(true);
  showTyping_();

  google.script.run
    .withSuccessHandler(handleAIResponse_)
    .withFailureHandler(handleError_)
    .askCipherAI(query, historyStr, MONTH_YEAR);
}

function handleAIResponse_(text) {
  setLoading_(false);
  removeTyping_();
  setTimeout(() => {
    appendAIMessage_(text || 'No response received.');
    chatHistory.push({ sender: 'AI', text: text });
    // Trim history
    if (chatHistory.length > 20) chatHistory = chatHistory.slice(-20);
  }, 180);
}

function handleError_(err) {
  setLoading_(false);
  removeTyping_();
  const msg = err?.message?.includes('API key') 
    ? 'Gemini API key not set. Use Budget Tools > Set Gemini API Key.'
    : err?.message?.includes('Supabase')
    ? 'Supabase credentials not set. Use Budget Tools > Set Supabase Credentials.'
    : 'Something went wrong. Check script logs.';
  appendError_(msg);
}

// ── Message renderers ─────────────────────────────────────────
function appendUserMessage_(text) {
  const el = document.createElement('div');
  el.className = 'msg user';
  el.innerHTML = `
    <div class="msg-bubble">${escHtml_(text)}</div>
    <span class="msg-time">${formatTime_()}</span>`;
  document.getElementById('chatbox').appendChild(el);
  scrollChat_();
}

function appendAIMessage_(text) {
  const el = document.createElement('div');
  el.className = 'msg ai';
  // Format numbers in the response with a subtle highlight
  const formatted = escHtml_(text).replace(/(\$[\d,]+\.?\d*)/g, '<strong>$1</strong>');
  el.innerHTML = `
    <div class="msg-bubble">${formatted}</div>
    <span class="msg-time">${formatTime_()}</span>`;
  document.getElementById('chatbox').appendChild(el);
  scrollChat_();
}

function appendError_(msg) {
  const el = document.createElement('div');
  el.className = 'msg-error';
  el.textContent = msg;
  document.getElementById('chatbox').appendChild(el);
  scrollChat_();
}

function showTyping_() {
  const el = document.createElement('div');
  el.id = 'typingBubble';
  el.className = 'typing-bubble';
  el.innerHTML = '<div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div>';
  document.getElementById('chatbox').appendChild(el);
  scrollChat_();
}

function removeTyping_() {
  const el = document.getElementById('typingBubble');
  if (el) el.remove();
}

function scrollChat_() {
  const box = document.getElementById('chatbox');
  setTimeout(() => { box.scrollTop = box.scrollHeight; }, 60);
}

// ── Header stats ──────────────────────────────────────────────
function loadHeaderStats_() {
  // Ask the AI for a structured summary to populate stats
  google.script.run
    .withSuccessHandler(function(text) {
      // Parse the AI's response for numbers - fallback to showing the month
      try {
        // Try to extract dollar amounts from any context we can get
        parseStatsFromResponse_(text);
      } catch(e) {
        clearStatLoaders_();
      }
    })
    .withFailureHandler(function() { clearStatLoaders_(); })
    .askCipherAI(
      'In exactly this format and nothing else, reply with three numbers separated by pipes: ' +
      'total income this month | total expenses this month | net savings this month. ' +
      'Use numbers only with dollar sign, like: $4200.00|$3100.00|$1100.00',
      '',
      MONTH_YEAR
    );
}

function parseStatsFromResponse_(text) {
  if (!text) { clearStatLoaders_(); return; }
  const parts = text.replace(/\s/g,'').split('|');
  if (parts.length >= 3) {
    setStatValue_('statIncome',   parts[0], 'positive');
    setStatValue_('statExpenses', parts[1], 'negative');
    const net = parseFloat(parts[2].replace(/[^0-9.-]/g, ''));
    setStatValue_('statNet', parts[2], net >= 0 ? 'positive' : 'negative');
  } else {
    clearStatLoaders_();
  }
}

function setStatValue_(id, value, cls) {
  const el = document.getElementById(id);
  el.textContent = value;
  el.className = 'stat-value ' + cls;
}

function clearStatLoaders_() {
  ['statIncome','statExpenses','statNet'].forEach(id => {
    const el = document.getElementById(id);
    el.textContent = '—';
    el.className = 'stat-value';
  });
}

// ── Summary panel ─────────────────────────────────────────────
function loadSummary_() {
  summaryLoaded = true;
  document.getElementById('summaryLoader').style.display = 'flex';
  document.getElementById('summaryContent').style.display = 'none';

  google.script.run
    .withSuccessHandler(renderSummary_)
    .withFailureHandler(function() {
      document.getElementById('summaryLoader').innerHTML =
        '<div style="color:var(--red);font-size:11px">Could not load budget data.</div>';
    })
    .askCipherAI(
      'List every budget category with its planned amount and actual spent amount this month. ' +
      'Format each line exactly as: CATEGORY|planned|actual — one per line, nothing else. ' +
      'Use numbers without dollar signs.',
      '',
      MONTH_YEAR
    );
}

function renderSummary_(text) {
  const loader  = document.getElementById('summaryLoader');
  const content = document.getElementById('summaryContent');
  loader.style.display = 'none';

  if (!text || !text.includes('|')) {
    content.innerHTML = '<div class="empty-state">No budget data found for ' + MONTH_YEAR + '.</div>';
    content.style.display = 'flex';
    return;
  }

  const lines = text.trim().split('\n').filter(l => l.includes('|'));
  let html = '<div class="section-head">Budget vs Actuals — ' + MONTH_YEAR + '</div>';

  lines.forEach(line => {
    const parts   = line.split('|');
    if (parts.length < 3) return;
    const cat     = parts[0].trim();
    const planned = parseFloat(parts[1]) || 0;
    const actual  = parseFloat(parts[2]) || 0;
    const pct     = planned > 0 ? Math.min((actual / planned) * 100, 100) : 0;
    const over    = actual > planned;
    const barCls  = actual > planned ? 'over' : pct > 80 ? 'warning' : 'ok';
    const amtCls  = over ? 'negative' : 'positive';

    html += `
      <div class="budget-row">
        <div class="budget-row-name">${escHtml_(cat)}</div>
        <div class="budget-row-amounts">
          <div class="budget-planned">/$${planned.toFixed(0)}</div>
          <div class="budget-actual ${amtCls}">$${actual.toFixed(0)}</div>
        </div>
        <div class="bar-wrap" style="grid-column:1/-1">
          <div class="bar-fill ${barCls}" style="width:${pct}%"></div>
        </div>
      </div>`;
  });

  html += `<button class="refresh-btn" onclick="summaryLoaded=false; loadSummary_()">Refresh data</button>`;
  content.innerHTML = html;
  content.style.display = 'flex';
}

// ── Goals panel ───────────────────────────────────────────────
function loadGoals_() {
  goalsLoaded = true;
  document.getElementById('goalsLoader').style.display = 'flex';
  document.getElementById('goalsContent').style.display = 'none';

  google.script.run
    .withSuccessHandler(renderGoals_)
    .withFailureHandler(function() {
      document.getElementById('goalsLoader').innerHTML =
        '<div style="color:var(--red);font-size:11px">Could not load goals.</div>';
    })
    .askCipherAI(
      'List all my savings goals. For each one, format exactly as: ' +
      'NAME|current_amount|target_amount|due_date — one per line, nothing else. ' +
      'Use numbers without dollar signs. Use "none" if no due date.',
      '',
      MONTH_YEAR
    );
}

function renderGoals_(text) {
  const loader  = document.getElementById('goalsLoader');
  const content = document.getElementById('goalsContent');
  loader.style.display = 'none';

  if (!text || !text.includes('|')) {
    content.innerHTML = '<div class="empty-state">No savings goals found.</div>';
    content.style.display = 'flex';
    return;
  }

  const lines = text.trim().split('\n').filter(l => l.includes('|'));
  let html = '<div class="section-head">Savings Goals</div>';

  lines.forEach(line => {
    const parts   = line.split('|');
    if (parts.length < 3) return;
    const name    = parts[0].trim();
    const current = parseFloat(parts[1]) || 0;
    const target  = parseFloat(parts[2]) || 1;
    const due     = parts[3]?.trim();
    const pct     = Math.min((current / target) * 100, 100);
    const pctDisp = pct.toFixed(1);
    const remaining = target - current;

    html += `
      <div class="goal-card">
        <div class="goal-top">
          <div class="goal-name">${escHtml_(name)}</div>
          <div class="goal-pct">${pctDisp}%</div>
        </div>
        <div class="goal-progress-bar">
          <div class="goal-bar-fill" style="width:${pct}%"></div>
        </div>
        <div class="goal-amounts">
          <span class="goal-current">$${current.toFixed(2)}</span>
          <span>$${target.toFixed(2)} target</span>
        </div>
        ${due && due !== 'none' ? `<div class="goal-due">Due: ${due} — $${remaining.toFixed(2)} remaining</div>` : ''}
      </div>`;
  });

  html += `<button class="refresh-btn" onclick="goalsLoaded=false; loadGoals_()">Refresh goals</button>`;
  content.innerHTML = html;
  content.style.display = 'flex';
}

function setLoading_(on) {
  isLoading = on;
  document.getElementById('sendBtn').disabled  = on;
  document.getElementById('userInput').disabled = on;
  if (!on) document.getElementById('userInput').focus();
}

function formatTime_() {
  const d = new Date();
  let h = d.getHours(), m = d.getMinutes().toString().padStart(2,'0');
  const ap = h >= 12 ? 'pm' : 'am';
  h = h % 12 || 12;
  return `${h}:${m} ${ap}`;
}

function getCurrentMonthYear_() {
  const d = new Date();
  return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
}

function escHtml_(str) {
  return String(str)
    .replace(/&/g,'&amp;')
    .replace(/</g,'&lt;')
    .replace(/>/g,'&gt;')
    .replace(/"/g,'&quot;');
}

function autoResizeTextarea_(el) {
  el.addEventListener('input', () => {
    el.style.height = 'auto';
    el.style.height = Math.min(el.scrollHeight, 80) + 'px';
  });
}
